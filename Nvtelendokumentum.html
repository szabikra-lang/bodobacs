<script>

/* ================= STATE ================= */

const GameState = {

  HOME:"home",

  GAME:"game",

  WIN:"win"

};

let state = GameState.HOME;

/* ================= CAMERA ================= */

let stream;

async function startCamera(){

  stream = await navigator.mediaDevices.getUserMedia({

    video:{ facingMode:"environment" }

  });

  const video = document.getElementById("cam");

  video.srcObject = stream;

  await video.play();

}

/* ================= ORIENTATION ================= */

let rotation = 0;

window.addEventListener("deviceorientation", e=>{

  rotation = e.alpha || 0;

});

/* ================= BUG SYSTEM ================= */

const bugs = [];

const bugImg = new Image();

bugImg.src = "data:image/png;base64,iVBORw0KGgoAAAANSUhEUgAAAAoAAAAKCAYAAACNMs+9AAAAFElEQVQYV2NkYGD4z0AEYBxVSFUAAN8ABQkR5XkAAAAASUVORK5CYII=";

function spawnBug(){

  bugs.push({

    worldX: Math.random()*360,

    size:40

  });

}

/* ================= GAME ================= */

let score = 0;

function update(){

  if(state !== GameState.GAME) return;

  if(bugs.length < 5){

    spawnBug();

  }

  updateRadar();

}

function render(){

  const c = arc;

  const ctx = c.getContext("2d");

  ctx.clearRect(0,0,c.width,c.height);

  bugs.forEach(b=>{

    const dx = b.worldX - rotation;

    const x = window.innerWidth/2 + dx*4;

    const y = window.innerHeight/2;

    ctx.drawImage(bugImg,x-20,y-20,40,40);

  });

}

/* ================= TAP ================= */

arc.addEventListener("click", e=>{

  const x = e.clientX;

  const y = e.clientY;

  for(let i=bugs.length-1;i>=0;i--){

    const dx = (window.innerWidth/2 + (bugs[i].worldX-rotation)*4) - x;

    if(Math.abs(dx)<30){

      bugs.splice(i,1);

      score++;

      document.getElementById("score").innerText=score;

      if(score>=10){

        alert("NYERTÉL!");

      }

    }

  }

});

/* ================= RADAR ================= */

function updateRadar(){

  if(!bugs.length) return;

  const diff = Math.abs(bugs[0].worldX - rotation);

  const percent = Math.max(0,100-diff);

  rfill.style.width = percent+"%";

}

/* ================= LOOP ================= */

function loop(){

  update();

  render();

  requestAnimationFrame(loop);

}

/* ================= CANVAS ================= */

function resize(){

  arc.width = window.innerWidth;

  arc.height = window.innerHeight;

}

window.addEventListener("resize",resize);

resize();

/* ================= START ================= */

document.getElementById("startBtn").onclick = async ()=>{

  await startCamera();

  // iOS motion permission

  if(typeof DeviceOrientationEvent.requestPermission==="function"){

    await DeviceOrientationEvent.requestPermission();

  }

  document.getElementById("home").style.display="none";

  state = GameState.GAME;

  loop();

};

/* ================= SAVE ================= */

function save(){

  localStorage.setItem("score",score);

}

function load(){

  const s = localStorage.getItem("score");

  if(s) score=parseInt(s);

}

load();

/* ================= SIMPLE MULTI ================= */

window.addEventListener("storage", e=>{

  if(e.key==="score"){

    score = parseInt(e.newValue);

    document.getElementById("score").innerText=score;

  }

});

setInterval(()=>localStorage.setItem("score",score),1000);
